create table users(
id serial primary key,
username varchar(50),
login varchar(120),
heshpassword varchar(255),
age integer,
role varchar(20) default 'user',
about varchar(255)
)
-- каждый автор это user но не каждый user author
create table author(
id serial primary key,
fk_author integer
)

create table video(
id serial primary key,
name_video varchar(255),
fk_author integer,
about_video varchar(255)
)

create table comment(
id serial primary key,
fk_users integer,
comm TEXT
)

alter table author 
add constraint fk_author_users
foreign key (fk_author) references users(id);

alter table video
add constraint fk_video_author
foreign key (fk_author) references author(id);

alter table comment 
add constraint fk_comment_users
foreign key (fk_users) references users(id)
